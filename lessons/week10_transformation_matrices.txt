glm::mat4 view(1);

glutIdleFunc(myIdle);


void myIdle()
{
	view = glm::rotate(view, 0.1f, glm::vec3(0, 1, 0));
	glUniformMatrix4fv(viewLoc, 1, GL_FALSE, glm::value_ptr(view));
	glutPostRedisplay();
}
